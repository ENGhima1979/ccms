import requests

API_KEY = "{{ api_key or 'YOUR_KEY' }}"
BASE    = "{{ base_url }}/api/v1"
HEADERS = {"X-API-Key": API_KEY}

# جلب المراسلات
resp = requests.get(f"{BASE}/correspondence", headers=HEADERS)
data = resp.json()
for item in data["items"]:
    print(item["ref_num"], item["subject"])

# إنشاء معاملة
new_corr = requests.post(f"{BASE}/correspondence", headers=HEADERS, json={
    "subject": "طلب عرض أسعار",
    "party": "شركة المقاولات العربية",
    "type": "out",
    "priority": "high"
})
print(new_corr.json())
const API_KEY = '{{ api_key or "YOUR_KEY" }}';
const BASE = '{{ base_url }}/api/v1';

// جلب المراسلات
const resp = await fetch(`${BASE}/correspondence`, {
  headers: { 'X-API-Key': API_KEY }
});
const { items } = await resp.json();
console.log(items);

// إنشاء معاملة جديدة
const create = await fetch(`${BASE}/correspondence`, {
  method: 'POST',
  headers: { 'X-API-Key': API_KEY, 'Content-Type': 'application/json' },
  body: JSON.stringify({
    subject: 'إشعار بالمستحقات المالية',
    party: 'وزارة المالية',
    type: 'out',
    priority: 'urgent'
  })
});
const result = await create.json();
console.log('Created:', result.ref_num);
